1 Introduction

Explore DoD2k v2.0 speleothem records from DoD2k.

DoD2k distinguishes archive (archiveType, e.g. Speleothem) from measurement (paleoData_proxy: δ18O, δ13C, growth rate, Mg/Ca). This notebook loads the compact CSV bundle, builds a long-format catalog, and plots an overview of speleothem sites.

2 Load DoD2k data

Data live in course/data/dod2k_v2.0/. The compact format stores one row per dataset; year and proxy values are comma-separated arrays in separate files.

Flow: compact CSV → wide table (load_dod2k_compact) → long catalog (load_dod2k_catalog) → RDS cache for fast reload.

2.2 Compact CSV loader

Three files are joined: metadata, year arrays, and proxy-value arrays.

Code 
#' Parse one line from a DoD2k compact array CSV (`id,val1,val2,…`).
#'
#' @param line One CSV line (without header).
#' @return List with `id` and numeric `values`.
parse_array_line <- function(line) {
  parts <- strsplit(line, ",", fixed = TRUE)[[1]]
  list(id = parts[[1]], values = as.numeric(parts[-1]))
}

#' Read a compact array column into a tibble.
#'
#' @param path Path to e.g. `dod2k_v2.0_compact_year.csv`.
#' @return Tibble: `dataset_id`, `values` (list of numeric vectors).
read_array_column <- function(path) {
  lines <- readLines(path, warn = FALSE)
  parsed <- lapply(lines[-1], parse_array_line)
  tibble(
    dataset_id = vapply(parsed, `[[`, character(1), "id"),
    values = lapply(parsed, `[[`, "values")
  )
}

#' Load DoD2k v2.0 compact bundle — one row per dataset, list-columns for time series.
#'
#' @return Wide tibble with metadata, `year`, and `paleoData_values`.
load_dod2k_compact <- function() {
  prefix <- basename(data_dir)
  meta <- read_csv(
    file.path(data_dir, paste0(prefix, "_compact_metadata.csv")),
    show_col_types = FALSE
  ) |>
    rename(dataset_id = any_of(c("datasetId", "...1")))

  years <- read_array_column(file.path(data_dir, paste0(prefix, "_compact_year.csv"))) |>
    rename(year = values)
  values <- read_array_column(
    file.path(data_dir, paste0(prefix, "_compact_paleoData_values.csv"))
  ) |>
    rename(paleoData_values = values)

  meta |>
    left_join(years, by = "dataset_id") |>
    left_join(values, by = "dataset_id")
}

2.3 Long-format catalog

Expands list-columns to one row per (dataset, year) and caches the result as RDS.

Code 
#' Convert compact data to a long catalog (0–2000 CE, finite values only).
#'
#' @param compact Output of [load_dod2k_compact()].
#' @return Tibble: `dataset_id`, `year`, `value`, `lon`, `lat`, `archive_type`, `proxy_type`.
compact_to_catalog <- function(compact) {
  rows <- vector("list", nrow(compact))
  n <- 0L
  for (i in seq_len(nrow(compact))) {
    rec <- compact[i, ]
    yrs  <- rec$year[[1]]
    vals <- rec$paleoData_values[[1]]
    if (!length(yrs) || !length(vals)) next

    ok <- yrs >= year_min & yrs <= year_max & is.finite(vals)
    if (!any(ok)) next

    n <- n + 1L
    rows[[n]] <- tibble(
      dataset_id = rec$dataset_id,
      year = as.integer(yrs[ok]),
      value = as.double(vals[ok]),
      lon = rec$geo_meanLon,
      lat = rec$geo_meanLat,
      archive_type = rec$archiveType,
      proxy_type = rec$paleoData_proxy
    )
  }
  if (!n) return(tibble())
  bind_rows(rows)
}

#' Read cached catalog from `output/cache/dod2k_timeseries.rds`.
#'
#' @return Tibble or `NULL` if cache does not exist.
read_catalog_cache <- function() {
  if (!file.exists(cache_rds)) return(NULL)
  readRDS(cache_rds) |>
    mutate(
      dataset_id = as.character(dataset_id),
      archive_type = as.character(archive_type),
      proxy_type = as.character(proxy_type),
      year = as.integer(year),
      value = as.double(value),
      lon = as.double(lon),
      lat = as.double(lat)
    )
}

#' Load or build the long-format catalog.
#'
#' @param refresh If `TRUE`, rebuild cache from CSV.
#' @return Long-format catalog tibble.
load_dod2k_catalog <- function(refresh = FALSE) {
  if (!refresh) {
    cached <- read_catalog_cache()
    if (!is.null(cached)) return(cached)
  }
  message("Building catalog from compact CSV …")
  catalog <- compact_to_catalog(load_dod2k_compact())
  saveRDS(catalog, cache_rds)
  catalog
}

3.2 Site tables

n_metadata = rows in the archive; n_with_data = series with values in 0–2000 CE.

Code 
#' One row per speleothem series with site metadata.
#'
#' @param catalog Long catalog from [load_dod2k_catalog()].
#' @param compact Wide compact table (for site names).
#' @return Tibble with `dataset_id`, coords, year range, `n_years`, `site`.
speleothem_sites <- function(catalog, compact) {
  sites <- catalog |>
    filter(archive_type == "Speleothem") |>
    group_by(dataset_id, proxy_type, lon, lat) |>
    summarize(
      year_start = min(year),
      year_end = max(year),
      n_years = n(),
      .groups = "drop"
    )

  meta <- compact |>
    filter(archiveType == "Speleothem") |>
    dplyr::select(dataset_id, geo_siteName) |>
    rename(site = geo_siteName)

  sites |>
    left_join(meta, by = "dataset_id") |>
    mutate(
      site = if_else(is.na(site) | site == "" | site == dataset_id, dataset_id, site),
      proxy_type = factor(proxy_type, levels = proxy_order)
    )
}

#' Count metadata rows vs. series with usable time series, by proxy type.
#'
#' @param compact Wide compact table.
#' @param catalog Long catalog.
#' @return Tibble: `proxy_label`, `proxy_type`, `n_metadata`, `n_with_data`.
speleothem_summary <- function(compact, catalog) {
  meta <- compact |>
    filter(archiveType == "Speleothem") |>
    count(paleoData_proxy, name = "n_metadata") |>
    rename(proxy_type = paleoData_proxy)

  with_data <- catalog |>
    filter(archive_type == "Speleothem") |>
    distinct(dataset_id, proxy_type) |>
    count(proxy_type, name = "n_with_data")

  meta |>
    full_join(with_data, by = "proxy_type") |>
    mutate(
      n_metadata = replace_na(n_metadata, 0L),
      n_with_data = replace_na(n_with_data, 0L),
      proxy_label = proxy_labels[proxy_type],
      proxy_type = factor(proxy_type, levels = proxy_order)
    ) |>
    arrange(proxy_type) |>
    transmute(proxy_label, proxy_type, n_metadata, n_with_data)
}

3.3 Site map

Interactive world map — colour and symbol by proxy type; hover shows site name and year range.

Code 
#' Interactive plotly map of speleothem sites.
#'
#' @param sites Output of [speleothem_sites()].
#' @return plotly object.
plot_speleothem_map <- function(sites) {
  sites <- sites |>
    mutate(
      proxy_label = proxy_labels[as.character(proxy_type)],
      hover = paste0(
        "<b>", site, "</b><br>", dataset_id, "<br>",
        "Proxy: ", proxy_label, "<br>",
        "Years: ", year_start, "–", year_end, " (n = ", n_years, ")"
      )
    )

  p <- plot_ly(type = "scattergeo", mode = "markers")
  for (px in proxy_order) {
    sub <- filter(sites, as.character(proxy_type) == px)
    if (!nrow(sub)) next
    p <- add_trace(
      p, data = sub, lon = ~lon, lat = ~lat, text = ~hover, hoverinfo = "text",
      name = paste0("Speleothem: ", proxy_labels[[px]], " (n=", nrow(sub), ")"),
      marker = list(
        symbol = proxy_symbols[[px]], size = 9,
        color = proxy_colors[[px]], line = list(width = 0.6, color = "#333")
      )
    )
  }
  p |> layout(
    title = list(
      text = paste0("<b>DoD2k speleothem archive</b><br><sup>", nrow(sites), " series</sup>"),
      x = 0.5, xanchor = "center"
    ),
    geo = list(
      projection = list(type = "natural earth"),
      showland = TRUE, landcolor = "#f0f0f0",
      showocean = TRUE, oceancolor = "#e6f2fa",
      showcountries = TRUE, countrycolor = "#bbbbbb",
      lataxis = list(range = c(-40, 75))
    ),
    legend = list(orientation = "h", y = -0.05, x = 0.5, xanchor = "center")
  )
}

3.4 Temporal coverage

How many series have at least one dated value per 100-year bin (actual years, not calendar span).

Code 
#' Stacked bar chart — series count per time bin.
#'
#' @param catalog Long-format catalog.
#' @param bin_width Bin width in years (default 100).
#' @return ggplot object.
plot_speleothem_density <- function(catalog, bin_width = 100) {
  density <- catalog |>
    filter(archive_type == "Speleothem") |>
    mutate(bin_start = (as.integer(year %/% bin_width)) * bin_width) |>
    distinct(dataset_id, proxy_type, bin_start) |>
    count(bin_start, proxy_type, name = "n_series") |>
    mutate(
      proxy_type = factor(proxy_type, levels = proxy_order),
      bin_mid = bin_start + bin_width / 2
    )

  ggplot(density, aes(bin_mid, n_series, fill = proxy_type)) +
    geom_col(width = bin_width * 0.9, colour = "white", linewidth = 0.2) +
    scale_fill_manual(values = proxy_colors, labels = proxy_labels[proxy_order], name = "Proxy type") +
    scale_x_continuous(limits = c(0, 2000), breaks = seq(0, 2000, 250), name = "Year (CE)") +
    labs(
      title = "Where observations exist",
      subtitle = "Stacked by proxy type (100-year bins)",
      y = "Series with data"
    ) +
    theme_clean()
}

## Long-format catalog

Expands list-columns to one row per `(dataset, year)` and caches the result as RDS.

```{r}
#' Convert compact data to a long catalog (0–2000 CE, finite values only).
#'
#' @param compact Output of [load_dod2k_compact()].
#' @return Tibble: `dataset_id`, `year`, `value`, `lon`, `lat`, `archive_type`, `proxy_type`.
compact_to_catalog <- function(compact) {
  rows <- vector("list", nrow(compact))
  n <- 0L
  for (i in seq_len(nrow(compact))) {
    rec <- compact[i, ]
    yrs  <- rec$year[[1]]
    vals <- rec$paleoData_values[[1]]

    ok <- yrs >= year_min & yrs <= year_max & is.finite(vals)
    if (!any(ok)) next

    n <- n + 1L
    rows[[n]] <- tibble(
      dataset_id = rec$dataset_id,
      year = as.integer(yrs[ok]),
      value = as.double(vals[ok]),
      lon = rec$geo_meanLon,
      lat = rec$geo_meanLat,
      archive_type = rec$archiveType,
      proxy_type = rec$paleoData_proxy
    )
  }
  if (!n) return(tibble())
  bind_rows(rows)
}

#' Read cached catalog from `output/cache/dod2k_timeseries.rds`.
#'
#' @return Tibble or `NULL` if cache does not exist.
read_catalog_cache <- function() {
  if (!file.exists(cache_rds)) return(NULL)
  readRDS(cache_rds) |>
    mutate(
      dataset_id = as.character(dataset_id),
      archive_type = as.character(archive_type),
      proxy_type = as.character(proxy_type),
      year = as.integer(year),
      value = as.double(value),
      lon = as.double(lon),
      lat = as.double(lat)
    )
}

#' Load or build the long-format catalog.
#'
#' @param refresh If `TRUE`, rebuild cache from CSV.
#' @return Long-format catalog tibble.
load_dod2k_catalog <- function(refresh = FALSE) {
  if (!refresh) {
    cached <- read_catalog_cache()
    if (!is.null(cached)) return(cached)
  }
  message("Building catalog from compact CSV …")
  catalog <- compact_to_catalog(load_dod2k_compact())
  saveRDS(catalog, cache_rds)
  catalog
}
```

```{r}
    left_join(meta, by = "dataset_id") |>
    mutate(
      site = if_else(is.na(site) | site == "" | site == dataset_id, dataset_id, site),
      proxy_type = factor(proxy_type, levels = proxy_order)
    )
}

#' Count metadata rows vs. series with usable time series, by proxy type.
#'
#' @param compact Wide compact table.
#' @param catalog Long catalog.
#' @return Tibble: `proxy_label`, `proxy_type`, `n_metadata`, `n_with_data`.
speleothem_summary <- function(compact, catalog) {
  meta <- compact |>
    filter(archiveType == "Speleothem") |>
    count(paleoData_proxy, name = "n_metadata") |>
    rename(proxy_type = paleoData_proxy)

  with_data <- catalog |>
    filter(archive_type == "Speleothem") |>
    distinct(dataset_id, proxy_type) |>
    count(proxy_type, name = "n_with_data")

  meta |>
    full_join(with_data, by = "proxy_type") |>
    mutate(
      n_metadata = replace_na(n_metadata, 0L),
      n_with_data = replace_na(n_with_data, 0L),
      proxy_label = proxy_labels[proxy_type],
      proxy_type = factor(proxy_type, levels = proxy_order)
    ) |>
    arrange(proxy_type) |>
    transmute(proxy_label, proxy_type, n_metadata, n_with_data)
}
```

## Site map

Interactive world map — colour and symbol by proxy type; hover shows site name and year range.

```{r}
#| label: speleothem-map-fn
#| code-fold: true

#' Interactive plotly map of speleothem sites.
#'
#' @param sites Output of [speleothem_sites()].
#' @return plotly object.
plot_speleothem_map <- function(sites) {
  sites <- sites |>
    mutate(
      proxy_label = proxy_labels[as.character(proxy_type)],
      hover = paste0(
        "<b>", site, "</b><br>", dataset_id, "<br>",
        "Proxy: ", proxy_label, "<br>",
        "Years: ", year_start, "–", year_end, " (n = ", n_years, ")"
      )
    )

  p <- plot_ly(type = "scattergeo", mode = "markers")
  for (px in proxy_order) {
    sub <- filter(sites, as.character(proxy_type) == px)
    if (!nrow(sub)) next
    p <- add_trace(
      p, data = sub, lon = ~lon, lat = ~lat, text = ~hover, hoverinfo = "text",
      name = paste0("Speleothem: ", proxy_labels[[px]], " (n=", nrow(sub), ")"),
      marker = list(
        symbol = proxy_symbols[[px]], size = 9,
        color = proxy_colors[[px]], line = list(width = 0.6, color = "#333")
      )
    )
  }
  p |> layout(
    title = list(
      text = paste0("<b>DoD2k speleothem archive</b><br><sup>", nrow(sites), " series</sup>"),
      x = 0.5, xanchor = "center"
    ),
    geo = list(
      projection = list(type = "natural earth"),
      showland = TRUE, landcolor = "#f0f0f0",
      showocean = TRUE, oceancolor = "#e6f2fa",
      showcountries = TRUE, countrycolor = "#bbbbbb",
      lataxis = list(range = c(-40, 75))
    ),
    legend = list(orientation = "h", y = -0.05, x = 0.5, xanchor = "center")
  )
}
```

## Temporal coverage

How many series have at least one dated value per 100-year bin (actual years, not calendar span).

```{r}
#| label: speleothem-density-fn
#| code-fold: true

#' Stacked bar chart — series count per time bin.
#'
#' @param catalog Long-format catalog.
#' @param bin_width Bin width in years (default 100).
#' @return ggplot object.
plot_speleothem_density <- function(catalog, bin_width = 100) {
  density <- catalog |>
    filter(archive_type == "Speleothem") |>
    mutate(bin_start = (as.integer(year %/% bin_width)) * bin_width) |>
    distinct(dataset_id, proxy_type, bin_start) |>
    count(bin_start, proxy_type, name = "n_series") |>
    mutate(
      proxy_type = factor(proxy_type, levels = proxy_order),
      bin_mid = bin_start + bin_width / 2
    )

  ggplot(density, aes(bin_mid, n_series, fill = proxy_type)) +
    geom_col(width = bin_width * 0.9, colour = "white", linewidth = 0.2) +
    scale_fill_manual(values = proxy_colors, labels = proxy_labels[proxy_order], name = "Proxy type") +
    scale_x_continuous(limits = c(0, 2000), breaks = seq(0, 2000, 250), name = "Year (CE)") +
    labs(
      title = "Where observations exist",
      subtitle = "Stacked by proxy type (100-year bins)",
      y = "Series with data"
    ) +
    theme_clean()
}
```
